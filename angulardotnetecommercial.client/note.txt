open the app.module.ts and import the FontAwesomeModule
import { FontAwesomeModule } from '@fortawesome/angular-fontawesome'
imports: [
    BrowserModule,

  ],


  import { faCoffee } from '@fortawesome/free-solid-svg-icons';
  Next, we create a variable name called faCoffee and assign the icon that we imported to the variable so can use it in the app.component.html. If we don’t do that, we can’t use it:

faCoffee = faCoffee;


  <div>
    <fa-icon ="faCoffee"></fa-icon>
</div>
